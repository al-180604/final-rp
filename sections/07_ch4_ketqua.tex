\section{Điều kiện chạy thử (Test Setup)}
\label{sec:ch4-test-setup}

\textbf{Hardware:}
\begin{itemize}
    \item 1x Coordinator: EFR32MG12 Wireless Starter Kit (BRD4162A)
    \item 1x Light Device: EFR32MG12 Radio Board (BRD4162A)
    \item 1x Switch Device: EFR32MG12 Radio Board (BRD4162A)
    \item 1x Occupancy Sensor: EFR32MG12 Radio Board (BRD4162A)
    \item Gateway: Ubuntu Linux (laptop / Raspberry Pi)
\end{itemize}

\textbf{Software:}
\begin{itemize}
    \item Simplicity Studio v5 with Gecko SDK 4.5.x
    \item MQTT Broker (Mosquitto / Cloud)
    \item Firebase (Realtime DB / Firestore)
    \item Flutter App (Android / iOS)
\end{itemize}

\textbf{Connectivity:}
\begin{itemize}
    \item UART: 115200 baud, 8N1
    \item MQTT: Cloud broker, QoS 0/1
    \item Zigbee: Channel 11, centralized Trust Center
\end{itemize}
